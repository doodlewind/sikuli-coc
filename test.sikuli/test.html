def collect():
    r.setAutoWaitTimeout(0.1)
    golds = r.findAll(Pattern().similar(0.55))
    if golds:
        for g in golds:
            r.click(g)

    elixirs = r.findAll()
    if elixirs:
        for e in elixirs:
            r.click(e)

    darkElixirs = r.findAll(Pattern().similar(0.80))
    reDarkElixirs = r.findAll(Pattern().similar(0.80).targetOffset(10,-2))
    if darkElixirs:
        for d in darkElixirs:
            r.click(d)


def startCOC():
    r.setAutoWaitTimeout(1)
    if r.exists():
        r.click()
    if r.wait():
        Debug.log("COCO Started")

        return True
    else:
        if r.exists():
            r.click()
        if r.exists():
            r.click()
        return False


def start():
    startFlag = False
    startCount = 0
    while not startFlag:
        startFlag = startCOC()
        startCount += 1
        if startCount == 10:
            break
    return startFlag
